536  void bubble_down(size_type _pos) {
537  size_type heap_size = heap_vec.size();
538  // positions of left, right nodes relative to parent in heap_vec
539  size_type left_child = 2 * _pos + 1;
540  size_type right_child = 2 * _pos + 2;
541  // reached end of heap, exit
542  if (left_child >= heap_size) return;
543  size_type min_pos = _pos;
544  // perform comparison between values of heap at _pos and left node pos
545  // and set min_pos to position of left node if comparison yields true
546  if (comp(heap_vec[_pos], heap_vec[left_child]))
547  min_pos = left_child;
548  // perform comparison between values of heap at min_pos and right node pos
549  // and set min_pos to position of right node if comparision yields true
550  if (right_child < heap_size && comp(heap_vec[min_pos], heap_vec[right_child]))
551  min_pos = right_child;
552  // if required, perform swap and bubble down from swapped min_pos
553  if (min_pos != _pos) {
554  std::swap(heap_vec[_pos], heap_vec[min_pos]);
555  bubble_down(min_pos);
556  }
557  }
567  void bubble_up(size_type _pos) {
568  // at top of heap, exit
569  if (!_pos) return;
570  // positon of parent node relative to a child node in heap_vec
571  size_type parent = (_pos - 1) / 2;
572  // perform comparison between heap values at parent and _pos
573  // and do swap and bubble up from parent if comparison is true
574  if (comp(heap_vec[parent], heap_vec[_pos])) {
575  std::swap(heap_vec[_pos], heap_vec[parent]);
576  bubble_up(parent);
577  }
578  }
585  void pop_top() noexcept {
586  size_type heap_size = heap_vec.size();
587  if (!heap_size) return;
588  // set top of heap to last heap element then remove last heap element
589  heap_vec[0] = std::move(heap_vec[--heap_size]);
590  heap_vec.pop_back();
591  // bubble down from top to get previously last heap element to correct position
592  bubble_down(0);
593  }
